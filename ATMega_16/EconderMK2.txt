/*********************************************
   FUNZIONE   : mcu_setup
   DESCRIZIONE: Inizializzazione periferiche MCU.
   PARAMETRI  : nessuno
   RETURN CODE: nessuno
*********************************************/
void mcu_setup(void)
{
	// Configurazione porte
	DDRA  = 0b11110000;	
	PORTA = 0b00001111;
	DDRB  = 0b00000000;
	PORTB = 0b11111111;
	DDRC  = 0b00000011;
	PORTC = 0b11111100;
	DDRD  = 0b01010000;
	PORTD = 0b10101111;
	
	MCUCR=0b00000011;
	SET(GICR,INT0);
	
	LCD_init();
	LCD_clear();
	LCD_home();
	LCD_cursor(1);
	
	TCCR0=0b00001011;
	OCR0=124;
	SET(TIMSK, OCIE0);
	
	//configurazione display
	

	// Configurazione periferiche utilizzate
	


	// watchdog
	wdt_reset();                   
	wdt_disable();        //disabilita wdt

	// interrupt
	SEI();                //abilita interruzioni
}

/*********************************************

int main(void)
{
	mcu_setup();
	char str1[20];
	char str2[20];
	
	while(1) 
	{
		
		if(timeout(2))
		{
			set_timer(2,1000);
			giri2=deltas/1024;
			giri3=((U32)(deltas*1000)/1024)%100;		//cambiare in u32 o 16
