/*********************************************
   FUNZIONE   : mcu_setup
   DESCRIZIONE: Inizializzazione periferiche MCU.
   PARAMETRI  : nessuno
   RETURN CODE: nessuno
*********************************************/
void mcu_setup(void)
{
	// Configurazione porte (esempio)
 	PORTB = 0b11111111;
	DDRB = 0b00000000;
	PORTC = 0b11110000;
	DDRC = 0b00001111;
	PORTD = 0b11111011;
	DDRD = 0b00000000;

	// watchdog
	wdt_reset();                   
	wdt_disable();        //disabilita wdt

	// interrupt
	SEI();                //abilita interruzioni
}
