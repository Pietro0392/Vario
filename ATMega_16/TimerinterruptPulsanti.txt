#include "common.h"

/*********************************************
   FUNZIONE   : mcu_setup
   DESCRIZIONE: Inizializzazione periferiche MCU.
   PARAMETRI  : nessuno
   RETURN CODE: nessuno
*********************************************/
void mcu_setup(void)
{
	// Configurazione porte
	DDRA  = 0b11110000;	
	PORTA = 0b00001111;
	DDRB  = 0b00000000;
	PORTB = 0b11111111;
	DDRC  = 0b00000000;
	PORTC = 0b11111111;
	DDRD  = 0b01010000;
	PORTD = 0b10101111;
	
	LCD_init();
	LCD_clear();
	LCD_home();
	LCD_cursor(1);
	
	TCCR0=0b00001011;
	OCR0=124;
	SET(TIMSK, OCIE0);
	
	//configurazione display
	

	// Configurazione periferiche utilizzate
	


	// watchdog
	wdt_reset();                   
	wdt_disable();        //disabilita wdt

	// interrupt
	SEI();                //abilita interruzioni
}


/*********************************************
   FUNZIONE   : main
   DESCRIZIONE: Main loop
   PARAMETRI  : nessuno
   RETURN CODE: ignorato
*********************************************/

volatile U16 millisec;
volatile U16 secondi;
volatile U16 minuti;
volatile int I = -1;
char inizio[17];

char stringatot[17];

ISR(TIMER0_COMP_vect)		// interrupt ogni millisecondo
{
	if(I==1 || I==-2)
	{
		
	millisec++;
	if(millisec==1000)
	{
		secondi++;
		millisec=0;
	}
	
	if(secondi==60)
	{
		minuti++;
		secondi=0;
	}
	}
}


/*

PB3 = Start
PB4 = Stop
PB5 = Reset
PB6 = Intertempo

*/

int main(void)
{
	mcu_setup();
	
	while(1) 
	{
			
			if(I==-1)
			{
			sprintf(inizio,"Premere PB3");
			LCD_writeln(0,inizio);
			}
		
		
		if(!TST(PINB,3))			//pstart premuto?
		{I=1;}
		
		
		if(!TST(PINB,5) && I==0)			//se il timer Ã¨ fermo e resettato
		{
			sprintf(inizio,"Premere PB3");
			LCD_writeln(0,inizio);
											millisec=0;
											minuti=0;
											secondi=0;
		}
		
		while(I==1)
		{
					if(!TST(PINB,4))			//pstop premuto?
					{
						I=0;
						LCD_writeln(0,stringatot);
						
					}
					
							if(!TST(PINB,5))			//reset premuto?
							{
								millisec=0;
								minuti=0;
								secondi=0;
							}
							
							if(!TST(PINB,6))
							{
								I=-2;
								
							}
							
		delayMs(100);
		if(secondi<10)
		{
			sprintf(stringatot,"0%d : 0%d : %d ",minuti,secondi,millisec);
		}
		if(secondi>=10)
		{
			sprintf(stringatot,"0%d : %d : %d ",minuti,secondi,millisec);
		}
		LCD_writeln(0,stringatot);
		}
	}
}
